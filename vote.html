/* ── 修改原有的提交函数 ── */ // 1. 提交投票 async function subVote() { if(!selId) return; const sb = document.getElementById('sb'); // 提交按钮 sb.disabled = true; sb.textContent = '呈递中…'; try { const r = await fetch('/api/session', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ action: 'vote', payload: { optionId: selId } }) }); const d = await r.json(); if(d.ok) { localStorage.setItem(KEY_C, selId); myChoice = selId; tick(); // 调用你原有的投票成功反馈 } else { throw new Error(); } } catch(e) { sb.disabled = false; sb.textContent = '确 认 选 择'; alert('网络异常，请重试'); } } // 2. 提交论点 async function subArg() { const el = document.getElementById('at'); // 输入框 const txt = el.value.trim(); if(txt.length < 10) return; const sb = document.getElementById('sb_arg') || document.querySelector('.arg-submit-btn'); // 适配你的按钮 if(sb) sb.disabled = true; try { await fetch('/api/session', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ action: 'submit_arg', payload: { text: txt } }) }); alert('论点已上呈！'); el.value = ''; // 触发你原有的成功样式 } catch(e) { alert('提交失败，请重试'); if(sb) sb.disabled = false; } }
